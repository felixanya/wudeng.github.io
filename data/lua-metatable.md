__index 值是一个table，如果找不到属性，就会去__index table里面找。这个行为一般被用来实现继承。

Animal = {}
Animal.prototype = {__index : {name : ""}}

Dog = {}
setmetatable(Dog, Animal.prototype)

function Animal:new(o)
  setmetatable(o, Animal.prototype)
  return o
end
